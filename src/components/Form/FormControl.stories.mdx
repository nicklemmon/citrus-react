# FormControl

## Used for the most common form interactions

Used to generate `<input/>` and `<select/>` elements.

```jsx
import React, { useState } from 'react'
import { FormControl } from '@citrus-react'

const MyComponent = () => {
  const [value, setValue] = useState('')

  return (
    <FormControl
      id="my-control"
      variant="text"
      value={value}
      onChange={e => setValue(e.target.value)}
    />
  )
}
```
