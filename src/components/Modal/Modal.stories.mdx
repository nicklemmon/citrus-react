# Modal

## Used for rendering overlays

```jsx
import { Modal, ModalHeader, ModalContent } from '@citrus/react'

const MyComponent = () => (
  <Modal>
    <ModalHeader>
      <ModalHeading>This is a modal</ModalHeading>
    </ModalHeader>
    <ModalContent>
      <p>The content goes here.</p>
    </ModalContent>
  </Modal>
)
```
